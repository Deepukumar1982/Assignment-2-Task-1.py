# Program to check if a number is even or odd

# Taking input from the user
number = int(input("Enter a number: "))

# Checking if the number is even or odd
if number % 2 == 0:
    print(number, "is an even number.")
else:
    print(number, "is an odd number.")


 output:
 Enter a number: 7
7 is an odd number.
